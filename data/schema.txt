Database schema for `data/database.sqlite`

Tables:

types
- uuid TEXT PRIMARY KEY NOT NULL
- type TEXT NOT NULL
- weight REAL NOT NULL

users
- uuid TEXT PRIMARY KEY NOT NULL
- username TEXT NOT NULL
- role TEXT NOT NULL

subjects
- uuid TEXT PRIMARY KEY NOT NULL
- name TEXT NOT NULL

predictedGrades
- predictedGradeID TEXT PRIMARY KEY NOT NULL
- userID TEXT NOT NULL REFERENCES users(uuid) ON DELETE CASCADE ON UPDATE CASCADE
- subjectID TEXT NOT NULL REFERENCES subjects(uuid) ON DELETE CASCADE ON UPDATE CASCADE
- score REAL NOT NULL

history
- historyEntryID TEXT PRIMARY KEY NOT NULL
- userID TEXT NOT NULL REFERENCES users(uuid) ON DELETE CASCADE ON UPDATE CASCADE
- subjectID TEXT NOT NULL REFERENCES subjects(uuid) ON DELETE CASCADE ON UPDATE CASCADE
- typeID TEXT NOT NULL REFERENCES types(uuid) ON DELETE CASCADE ON UPDATE CASCADE
- score REAL NOT NULL
- studied_at DATETIME NOT NULL
